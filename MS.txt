// Each timeframe now has its own colour so you can tell the rectangles apart:
//   4H zones  = BLUE
//   30m zones = ORANGE
//   5m zones  = PURPLE
// Within each timeframe:
//   Bearish (break-down) zone = richer fill + SOLID border
//   Bullish (reclaim)   zone = lighter fill + DASHED border
// Structure rules (same on every timeframe):
// 1. A higher high creates a bear zone from the last bearish candle before that HH
//    (that candle's close down to its lower wick). A close below it flips bearish.
// 2. The last bullish candle before a bearish break becomes the bull reclaim zone
//    (that candle's close up to its upper wick). A bullish close above it flips bullish.
// 3. While bearish, each fresh lower low updates the bull zone.
// Signals/alerts keep the original logic: 4H bearish -> 5m sell breaks of 30m bear zones;
// 4H bullish -> 5m buy breaks of 30m bull zones. Designed to run on a 5-minute chart.

f_lastCandleZone(bool wantBullish) =>
    float zoneTop = na
    float zoneBottom = na
    int zoneTime = na
    bool found = false

    for i = 1 to zoneSearchBars by 1
        bool candleMatches = wantBullish ? close[i] > open[i] : close[i] < open[i]
        if not found and candleMatches
            zoneTop := wantBullish ? high[i] : close[i]
            zoneBottom := wantBullish ? close[i] : low[i]
            zoneTime := time[i]
            found := true
            found

    [zoneTop, zoneBottom, zoneTime]

f_marketStructure() =>
    var int trend = 0
    var float bearZoneTop = na
    var float bearZoneBottom = na
    var int bearZoneCandleTime = na
    var float bullZoneTop = na
    var float bullZoneBottom = na
    var int bullZoneCandleTime = na

    int newBearZoneEvent = na
    int newBearZoneEventClose = na
    int newBullZoneEvent = na
    int newBullZoneEventClose = na
    int bearishFlipEvent = na
    int bearishFlipEventClose = na
    int bullishFlipEvent = na
    int bullishFlipEventClose = na

    float previousHighs = ta.highest(high[1], hhLookbackBars)
    float previousLows = ta.lowest(low[1], llLookbackBars)
    bool madeHigherHigh = not na(previousHighs) and high > previousHighs
    bool madeLowerLow = not na(previousLows) and low < previousLows
    bool bearishFlip = trend == 1 and not na(bearZoneBottom) and close < bearZoneBottom
    bool bullishFlip = trend == -1 and not na(bullZoneTop) and close > open and close > bullZoneTop

    if bearishFlip
        trend := -1
        bearishFlipEvent := time
        bearishFlipEventClose := time_close

        [lastBullTop, lastBullBottom, lastBullTime] = f_lastCandleZone(true)
        if not na(lastBullTop)
            bullZoneTop := lastBullTop
            bullZoneBottom := lastBullBottom
            bullZoneCandleTime := lastBullTime
            newBullZoneEvent := time
            newBullZoneEventClose := time_close
            newBullZoneEventClose

    else if bullishFlip
        trend := 1
        bullishFlipEvent := time
        bullishFlipEventClose := time_close

        [lastBearTopAfterFlip, lastBearBottomAfterFlip, lastBearTimeAfterFlip] = f_lastCandleZone(false)
        if not na(lastBearTopAfterFlip)
            bearZoneTop := lastBearTopAfterFlip
            bearZoneBottom := lastBearBottomAfterFlip
            bearZoneCandleTime := lastBearTimeAfterFlip
            newBearZoneEvent := time
            newBearZoneEventClose := time_close
            newBearZoneEventClose

    else if trend >= 0 and madeHigherHigh
        [lastBearTopOnHh, lastBearBottomOnHh, lastBearTimeOnHh] = f_lastCandleZone(false)
        if not na(lastBearTopOnHh) and (na(bearZoneCandleTime) or lastBearTimeOnHh != bearZoneCandleTime)
            trend := 1
            bearZoneTop := lastBearTopOnHh
            bearZoneBottom := lastBearBottomOnHh
            bearZoneCandleTime := lastBearTimeOnHh
            newBearZoneEvent := time
            newBearZoneEventClose := time_close
            newBearZoneEventClose

    else if trend == -1 and madeLowerLow
        [lastBullTopOnLl, lastBullBottomOnLl, lastBullTimeOnLl] = f_lastCandleZone(true)
        if not na(lastBullTopOnLl) and (na(bullZoneCandleTime) or lastBullTimeOnLl != bullZoneCandleTime)
            bullZoneTop := lastBullTopOnLl
            bullZoneBottom := lastBullBottomOnLl
            bullZoneCandleTime := lastBullTimeOnLl
            newBullZoneEvent := time
            newBullZoneEventClose := time_close
            newBullZoneEventClose

    [trend, bearZoneTop, bearZoneBottom, bearZoneCandleTime, bullZoneTop, bullZoneBottom, bullZoneCandleTime, newBearZoneEvent, newBearZoneEventClose, newBullZoneEvent, newBullZoneEventClose, bearishFlipEvent, bearishFlipEventClose, bullishFlipEvent, bullishFlipEventClose]

f_trimBoxes() =>
    while array.size(zoneBoxes) > maxHistoricalZones
        box oldestBox = array.shift(zoneBoxes)
        box.delete(oldestBox)

// A box is only archived (and eligible for trimming) once it stops being active.
// Active zones are held in their own variables and are never trimmed.
f_stopBox(box zoneBox, int stopTime) =>
    if not na(zoneBox)
        box.set_extend(zoneBox, extend.none)
        box.set_right(zoneBox, stopTime)
        array.push(zoneBoxes, zoneBox)
        f_trimBoxes()

f_makeZoneBox(int leftTime, int rightTime, float topPrice, float bottomPrice, color fillColor, color borderColor, int borderWidth, string borderStyle, string tfLabel) =>
    box newBox = na
    if not na(leftTime) and not na(rightTime) and not na(topPrice) and not na(bottomPrice)
        newBox := box.new(left = leftTime, top = topPrice, right = rightTime, bottom = bottomPrice, xloc = xloc.bar_time, extend = extend.right, bgcolor = fillColor, border_color = borderColor, border_width = borderWidth, border_style = borderStyle, text = tfLabel, text_size = size.tiny, text_halign = text.align_left, text_valign = text.align_top, text_color = borderColor)
        newBox
    newBox
